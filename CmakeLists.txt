cmake_minimum_required(VERSION 3.20)
project(nxruby C)

if(MSVC)
    add_compile_options(/utf-8)
endif()

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED True)
set(CMAKE_C_EXTENSIONS OFF)

# 依存ライブラリのパスの設定
set(MRUBY_DIR "${CMAKE_CURRENT_SOURCE_DIR}/deps/mruby")
set(SDL3_DIR  "${CMAKE_CURRENT_SOURCE_DIR}/deps/SDL3")
# 画像・フォント・音声のパスも追加
set(SDL3_IMAGE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/deps/SDL3_image")
set(SDL3_TTF_DIR   "${CMAKE_CURRENT_SOURCE_DIR}/deps/SDL3_ttf")
set(SDL3_MIXER_DIR "${CMAKE_CURRENT_SOURCE_DIR}/deps/SDL3_mixer")

# 実行ファイルとソースファイルの設定
# ※src配下のファイルをリストアップ（先ほどのmain.cなどのファイル名に合わせています）
add_executable(nxruby 
    src/main.c
    src/window.c
    src/input.c
    src/image.c
    src/sprite.c
    src/sound.c
    src/font.c
)

# インクルードパスの設定
target_include_directories(nxruby PRIVATE 
    src
    include
    "${MRUBY_DIR}/include"
    "${SDL3_DIR}/include"
    "${SDL3_IMAGE_DIR}/include"
    "${SDL3_TTF_DIR}/include"
    "${SDL3_MIXER_DIR}/include"
)

# ライブラリのリンク
if(WIN32)
    target_link_libraries(nxruby PRIVATE
        # --- mruby ---
        "${MRUBY_DIR}/lib/libmruby.lib" 

        # --- SDL3本体と拡張 ---
        "${SDL3_DIR}/lib/x64/SDL3.lib"
        "${SDL3_IMAGE_DIR}/lib/x64/SDL3_image.lib"
        "${SDL3_TTF_DIR}/lib/x64/SDL3_ttf.lib"
        "${SDL3_MIXER_DIR}/lib/x64/SDL3_mixer.lib"

        # --- Windowsシステムライブラリ（nxruby_build.rbにあったもの） ---
        ws2_32
        user32
        shell32
        advapi32
        setupapi
        gdi32
        imm32
        ole32
        oleaut32
        version
    )
endif()

# --- ビルド後の自動コピー処理 ---

# 1. 実行ファイル (nxruby.exe) を プロジェクト直下の bin/ にコピー
add_custom_command(TARGET nxruby POST_BUILD
    # nxruby.exeをコピー (すでに同じファイルがあればスキップ)
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        $<TARGET_FILE:nxruby>
        "C:/Dev/nxruby_app"
)
