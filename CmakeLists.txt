cmake_minimum_required(VERSION 3.20)
project(nxruby C)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED True)
set(CMAKE_C_EXTENSIONS OFF)

# 依存ライブラリのパスの設定
set(MRUBY_DIR "${CMAKE_CURRENT_SOURCE_DIR}/deps/mruby")
set(SDL3_DIR  "${CMAKE_CURRENT_SOURCE_DIR}/deps/SDL3")

# 実行ファイルとソースファイルの設定
add_executable(nxruby 
    src/main.c
    src/nx_window.c
    src/nx_input.c
)

# インクルードパスの設定
target_include_directories(nxruby PRIVATE 
    src
    include
    "${MRUBY_DIR}/include"
    "${SDL3_DIR}/include"
)

# ライブラリのリンク
if(WIN32)
    target_link_libraries(nxruby PRIVATE
        "${MRUBY_DIR}/lib/libmruby.lib"
        "${SDL3_DIR}/lib/x64/SDL3.lib"
        ws2_32
    )
endif()

# DLLのコピー設定 (実行時に SDL3.dll が必要だから)
add_custom_command(TARGET nxruby POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    "${SDL3_DIR}/lib/x64/SDL3.dll"
    "$<TARGET_FILE_DIR:nxruby>"
    COMMENT "Copying SDL3.dll to build directory..."
)
